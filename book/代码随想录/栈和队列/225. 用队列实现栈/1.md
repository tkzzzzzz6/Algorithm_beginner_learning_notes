使用队列,实现栈的功能

方法 1:两个队列,一个队列用来操作,另一个队列用来备份
```cpp
/*

class MyStack {
public:
    queue<int> q1,q2; //q2用于备份
    MyStack() {
        
    }
    
    void push(int x) {
        q1.push(x);
    }
    
    int pop() {
        int n = q1.size();
        if(n < 1)return -1;
        --n;
        while(n--){
            q2.push(q1.front());
            q1.pop();
        }

        int res = q1.front();
        q1.pop();

        q1 = q2;
        while(!q2.empty())q2.pop();
        return res;
    }
    
    int top() {
        int n = q1.size();
        if(n < 1)return -1;
        --n;
        while(n--){
            q2.push(q1.front());
            q1.pop();
        }

        int res = q1.front();
        q2.push(q1.front());

        q1 = q2;
        while(!q2.empty())q2.pop();
        return res;
    }
    
    bool empty() {
        return q1.empty();
    }
};
```
