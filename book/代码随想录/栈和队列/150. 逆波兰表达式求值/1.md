1. 什么是逆波兰表达式
前缀/中缀/后缀表达式

逆波兰表达式属于后缀表达式

上面的表达式中,前缀,中缀,后缀表达式分别是对应二叉树的前序遍历,中序遍历,后序遍历

后缀表达式的优点:不需要关系括号,不需要考虑运算符优先级,从左到右依次计算即可

```cpp
/*
 * @lc app=leetcode.cn id=150 lang=cpp
 * @lcpr version=30204
 *
 * [150] 逆波兰表达式求值
 */
class Solution {
public:

    int evalRPN(vector<string>& tokens) {
        stack<int> st;
        for(auto c:tokens){
            if(c != "+" && c != "-" && c != "*" && c != "/"){
                st.push(stoi(c));
            }else{
                int b = st.top();
                st.pop();
                int a = st.top();
                st.pop();
                if(c == "+"){
                    st.push(a+b);
                }else if(c == "-"){
                    st.push(a-b);
                }else if(c == "*"){
                    st.push(a*b);
                }else{
                    st.push(a/b);
                }
            }
        }
        return st.top();
    }
};
```
