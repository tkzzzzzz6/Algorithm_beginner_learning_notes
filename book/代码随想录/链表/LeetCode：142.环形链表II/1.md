方法1:快慢指针

具体做法:定义快指针每次走2个节点,定义慢指针每次走1个节点,如果链表没有环的话,两个指针不会相遇,如果有环的话,快指针比慢指针在环内按照移动一个节点的速度靠近慢指针,最后两个指针一定会相遇

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
  ListNode *detectCycle(ListNode *head) {
    ListNode *fast = head;
    ListNode *slow = head;
    while (fast != nullptr && fast->next != nullptr) {
      fast = fast->next->next;
      slow = slow->next;
      if (fast == slow) {
        ListNode *index1 = head;
        ListNode *index2 = fast;
        while (index1 != index2) {
          index1 = index1->next;
          index2 = index2->next;
        }
        return index1;
      }
    }
    return NULL;
  }
};

```

方法2:哈希表

定义一个哈希表,每次遍历的时候判断当前节点是否在哈希表中,不在的话添加到哈希表中,在的话说明有环,返回当前节点

```cpp
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        unordered_set<ListNode* > visited;
        while(head != nullptr){
            if(visited.count(head)){
                return head;
            }
            visited.insert(head);
            head = head->next;
        }
        return NULL;
    }
};

```
