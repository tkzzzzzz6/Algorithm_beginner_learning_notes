哈希表的适用场景,给一个元素,判断这个元素是否在集合中出现过
方法 1:如果对数字大小没有要求,需要使用 set 哈希

cpp 中有 3 个 set,分别是

- set:红黑树实现
- unordered_set:哈希映射实现,做映射和取值的效率最高,不重复
- multiset:红黑树实现,可以重复

```cpp
/*
 * @Author: tkzzzzzz6
 * @Date: 2026-04-11 09:39:48
 * @LastEditors: tkzzzzzz6
 * @LastEditTime: 2026-04-11 09:42:55

class Solution {
public:
  vector<int> intersection(vector<int> &nums1, vector<int> &nums2) {
    unordered_set<int>result;
    unordered_set<int>nums_set(nums1.begin(),nums1.end());
    for(auto x:nums2){
        if(nums_set.find(x) != nums_set.end()){
            result.insert(x);
        }
    }
    return vector<int>(result.begin(),result.end());
  }
};

```

方法 2:如果对数字大小有限制,可以使用数组哈希,速度会快一点

```cpp
class Solution {
public:
  vector<int> intersection(vector<int> &nums1, vector<int> &nums2) {
    unordered_set<int> result; // 自动去重
    int hash[10001] = {0}; // 题目说了num1/2中元素的大小在 0- 1000
    for (auto x : nums1) {
      hash[x] = 1;
    }
    for (auto x : nums2) {
      if (hash[x] == 1)
        result.insert(x);
    }
    return vector<int>(result.begin(), result.end());
  }
};

```
