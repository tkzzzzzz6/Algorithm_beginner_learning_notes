
方法1:map哈希

创建一个 unordered_map,存储 A+B 的出现次数,然后遍历 C+D 的组合,如果 A+B+C+D=0,说明 A+B=-C-D,如果 -C-D 在 map 中出现过,说明存在满足条件的组合,count+=map[-C-D]

```cpp
class Solution {
public:
  int fourSumCount(vector<int> &nums1, vector<int> &nums2, vector<int> &nums3,vector<int> &nums4) {
    unordered_map<int, int> umap;
    int count = 0;
    for (auto a : nums1) {
      for (auto b : nums2) {
        umap[a + b]++;
      }
    }

    for (auto c : nums3) {
      for (auto d : nums4) {
        if (umap.find(-c - d) != umap.end()) {
          count += umap[-c - d];
        }
      }
    }
    return count;
  }
};

```
