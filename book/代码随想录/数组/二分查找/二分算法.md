数组有序,无重复

两种写法,左闭右闭,左闭右开

1. 左闭右闭
```cpp
r = nums.size() - 1
while(l <= r)
if(nums[mid] > target)
    r = mid - 1
```
```cpp
class Solution{
public:
    int search(vector<int>&nums,int target){
        int l = 0,r = nums.size() - 1;
        while(l <= r){
            int mid = l + ((r - l) >> 1); //防止 l+r过大
            if(nums[mid] < target)
                l = mid + 1;
            else if(nums[mid] > target)
                r = mid - 1;
            else
                return mid;
        }
        return -1;
    }
};
```

1. 左闭右开
```cpp
r = nums.size()
while(l < r)
if(nums[mid] > target)
    r = mid - 1
```

```cpp
class Solution{
public:
    int search(vector<int>&nums,int target){
        int l = 0,r = nums.size();
        while(l < r){
            int mid = l + ((r - l) >> 1); //防止 l+r过大
            if(nums[mid] < target)
                l = mid + 1;
            else if(nums[mid] > target)
                r = mid;
            else
                return mid;
        }
        return -1;
    }
};
```
