# 方法一 二维前缀和

sum数组比grid数组大一圈,

```cpp
// 二维前缀和
class Solution {
public:
    int countSubmatrices(vector<vector<int>>& grid, int k) {
        int m = grid.size(),n = grid[0].size();
        int ans = 0;
        vector<vector<int>> sum(m+1,vector<int>(n+1));
        
        for(int i = 0;i<m;++i){
            for(int j = 0;j<n;++j){
                sum[i+1][j+1] = sum[i][j+1] + sum[i+1][j] - sum[i][j] + grid[i][j];
                ans += sum[i+1][j+1] <= k;
            }
        }
        return ans;
    }
};

```

## 拓展:原地二维前缀和
直接在grid数组上进行操作
```cpp
// 原地二维前缀和
class Solution {
public:
  int countSubmatrices(vector<vector<int>> &grid, int k) {
    int m = grid.size(), n = grid[0].size();
    int ans = 0;

    for (int i = 0; i < m; ++i) {
      for (int j = 0; j < n; ++j) {
        if(i > 0)grid[i][j] += grid[i-1][j];
        if(j > 0)grid[i][j] += grid[i][j-1];
        if(i >0 && j > 0) grid[i][j] -= grid[i-1][j-1];

        ans += (grid[i][j] <= k);
      }
    }
    return ans;
  }
};

```

# 方法二 维护每列的元素和
从左到右一行一行扫描判断
```cpp
// 维护每列元素和
class Solution {
public:
  int countSubmatrices(vector<vector<int>> &grid, int k) {
    int n = grid[0].size();
    vector<int> col_sum(n);
    int ans = 0;
    for(auto &row:grid){
        int s = 0;
        for(int j = 0;j<n;++j){
            col_sum[j] += row[j];
            s += col_sum[j];
            if(s > k)break;
            ++ans;
        }
    }
    return ans;
  }
};

```
