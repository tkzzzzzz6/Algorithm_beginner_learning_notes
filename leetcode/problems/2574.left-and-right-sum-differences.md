# [2574] 左右元素和的差值


class Solution:
    def leftRightDifference(self, nums: List[int]) -> List[int]:
        res = []
        total = sum(nums)
        leftsum = 0
        for i in range(len(nums)):
            res.append(abs(total - 2*leftsum - nums[i]))
            leftsum += nums[i]

        return res
