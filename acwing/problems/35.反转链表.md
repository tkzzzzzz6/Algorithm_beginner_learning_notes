解法 1:迭代

```cpp
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(!head || !head->next)return head;
        auto a = head,b = head->next;
        
        while(b){
            auto t = b->next;
            b->next = a;
            a = b,b = t;
        }

        head->next = nullptr;

        return a;
    }
};
```

解法 2:递归

```cpp
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(!head || !head->next)return head; //这里可以理解为递归的边界

        auto tail = reverseList(head->next);

        head->next->next = head; //回溯的时候让后一个节点指向当前节点,相当于递归的实际操作部分
        head->next = nullptr;

        return tail;
    }
};
```
