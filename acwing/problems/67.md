方法 1:手动实现二分:

```cpp
class Solution {
   public:
    int getNumberOfK(vector<int> &nums, int k) {
        if (nums.empty())
            return 0;

        int l = 0, r = nums.size();
        // 找第一个 k 的位置
        while (l < r) {
            int mid = (l + r) >> 1;
            if (nums[mid] < k)
                l = mid + 1;
            else
                r = mid;
        }
        if (nums[l] != k)
            return 0;

        // 找第一个大于 k 的数的位置
        int left = l;
        r = nums.size(); // nums[l] == k,l的值可以不修改
        while (l < r) {
            int mid = (l + r) >> 1;
            if (nums[mid] <= k)
                l = mid + 1;
            else
                r = mid;
        }

        return l - left;
    }
};
```

方法 2:库函数调用

```cpp
class Solution {
   public:
    int getNumberOfK(vector<int> &nums, int k) {
        if (nums.empty())
            return 0;

        // 第一个 >= k 的位置
        auto pos_left = lower_bound(nums.begin(), nums.end(), k);
        // 第一个 > k 的位置
        auto pos_right = upper_bound(nums.begin(), nums.end(), k);

        return static_cast<int>(pos_right - pos_left);
    }
};
```
