约瑟夫问题

# 思路 1:队列实现

## 使用 List
可以理解为原地删除,效率比用 queue 等容器高
```cpp
#include <list>
class Solution {
   public:
    int lastRemaining(int n, int m) {
        list<int> q;
        for (int i = 0; i < n; ++i)
            q.push_back(i);
        auto it = q.begin();
        int k = m - 1;
        while (q.size() > 1) {
            while (k--) {
                ++it;
                if (it == q.end())
                    it = q.begin();
            }
            it = q.erase(it);
            if (it == q.end())
                it = q.begin();
            k = m - 1;
        }

        return q.front();
    }
};
```

## 使用 queue
不删除的元素重新入队,搬移元素,效率会低一点
```cpp
#include <list>
class Solution {
   public:
    int lastRemaining(int n, int m) {
        queue<int> q;
        for (int i = 0; i < n; ++i)
            q.push(i);
        int k = 0;
        while (q.size() > 1) {
            if (k == m - 1) {
                q.pop();
                k = 0;
            }
            ++k;
            q.push(q.front());
            q.pop();
        }

        return q.front();
    }
};
```

# 思路 2:递归实现

![1778131554846.png](https://tk-pichost-1325224430.cos.ap-chengdu.myqcloud.com/blog/1778131554846.png)

后一轮的起始点的原始索引 $j$ 和本轮起始点原始索引 $i$ 的关系

$$
\begin{cases}
n = n -1 \\
j = (i+m) \mod n
\end{cases}
$$

```cpp
class Solution {
   public:
    int lastRemaining(int n, int m) {
        if (n == 1)
            return 0;
        return (lastRemaining(n - 1, m) + m) % n;
    }
};
```
