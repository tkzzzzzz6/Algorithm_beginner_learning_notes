# 思路
暴力枚举每个柱子,向左右两边扩展,直到遇到比当前柱子矮的柱子为止,计算矩形面积,更新最大值
时间复杂度:O(n^2)

```cpp
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 1010;
vector<int> h(N);

int main(){
    int n;cin >> n;
    for(int i = 1;i<=n;++i)cin >> h[i];

    int res = 0;
    for(int i = 1;i<=n;++i){
        int left = i,right = i;
        while(left >= 1 && h[left] >= h[i])--left;
        while(right <= n && h[right] >= h[i])++right;
        res = max(res,(right - left - 1)*h[i]);
    }

    cout << res << endl;

    return 0;
}
```
